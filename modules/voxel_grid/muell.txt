template <size_t SIZE>
class ChunkMap {
    static_assert(MAX_GRID_SIZE*2 < SIZE); // siehe 'next_idx'
    friend struct UnLoadData;

    static constexpr ushort SEARCH_DEPTH = 13;
    static constexpr ushort TRYS = 20;
    size_t _occupancy = 0;
    
    std::array<Chunk, SIZE> chunks{};
    
    inline static size_t hash(const ChunkKey& key) {
        size_t id_hash = std::hash<size_t>{}(key.grid_id);
        auto pre_hash = static_cast<size_t>(
              key.koords.x * SEARCH_DEPTH
            + key.koords.y * MAX_GRID_SIZE
            + key.koords.z * MAX_GRID_SIZE * MAX_GRID_SIZE)
            ^ id_hash;
        return (pre_hash + (pre_hash==0)) % SIZE; // TODO(erledigt! ich lass es trotzdem mal noch stehen...): "Remainder by zero is undefined" vieleicht muss ich das verbessern
    }

    // TODO: Was ist besser:
    //      - kein Clustering
    //      - Chash-optimiert, aber mit möglicher Clusterbildung
    // TODO: Messe Kollisionen -> ERR_PRINT_ED wenn Kollision oder Counter
    inline static void next_idx(ushort &count, size_t &idx, const ChunkKey &key) {
        count += 1;
        idx += static_cast<size_t>(count * 2 + key.koords.y);
        idx %= SIZE;
    }
    // inline static void next_idx_simple(ushort &count, size_t &idx, const ChunkKey &key) {
    //     count += 1;
    //     idx += 1;
    //     if (unlikely(SIZE <= idx)) {
    //         idx -= SIZE;
    //     }
    // }

public:
    //TODO: entferne diesen Rotz!
    String msg = "";

    inline double get_occupancy_rate() {
        return static_cast<double>(_occupancy) / static_cast<double>(SIZE);
    }

    Chunk *get(const ChunkKey &key) {
        ushort count = 0;
        size_t idx = hash(key);
        while (count < SEARCH_DEPTH) {
            if (key == chunks[idx] && !chunks[idx].is_free()) {
                return &chunks[idx];
            }
            next_idx(count, idx, key);
        }
        return nullptr;
    }

    Chunk steal(const ChunkKey &key) {
        Chunk *chunk = get(key);
        if (chunk) {
            _occupancy -= 1;
            chunk->free();
        }
        return *chunk;
    }
    void remove(const ChunkKey &key) {
        Chunk *chunk = get(key);
        if (chunk) {
            _occupancy -= 1;
            chunk->free();
        }
    }

    void remove(Chunk *chunk) {
        ERR_FAIL_COND_MSG(chunk->is_free(), "Chunk ist bereits gelöscht!");
        _occupancy -= 1;
        chunk->free();
    }

    Chunk *push(ChunkKey &&key) {
        msg = "";
        // TODO: dieser Teil erstmal ein workaround und zum sicher gehen:
        if (SIZE * 0.8 < _occupancy) {
            return nullptr; // mehr als 80% sollte nicht befüllt werden
        }
        // // überprüfen ob Chunk breits geladen ist
        // if (auto r = get(key)) {
        //     return r; // TODO: sollte vor 'push' geprüft werden
        // }
        
        // key zu index auflösen
        size_t idx = hash(key);
        // merke den ersten index, beim Scheitern wird diese Var. genutzt um das zuverschiebene Element zu bestimmen
        size_t move_idx = idx;
        // nutze loop um 'SEARCH_DEPTH' viele Kollisionen aufzulösen
        ushort count = 0;
        while (count < SEARCH_DEPTH) {
            if (chunks[idx].is_free()) {
                chunks[idx] << key;
                _occupancy += 1;
                return &chunks[idx];
            }
            next_idx(count, idx, key);
        }
        // Wenn nach 'SEARCH_DEPTH' Kollisionen kein Platz gefunden wurde: Chunk austauschen und neu Versuchen (hierbei MÜSSEN RIDs getauscht werden!! )
        msg = "Oh Nyo! Es hat nicht geklappt OwO'";
        // Daten des zu überschreibenen Chunks kopieren
        Chunk moving_chunk{chunks[move_idx]};
        // Chunk, der keinen Platz gefunden hat, einsetzen (RIDs fehlen hier noch!)
        chunks[move_idx] << key;
        // Funktion wiederholen, jetzt mit herausgepflückten Chunk, zurück kommen die freien RIDs
        // nutze rekursion um mehrere Versuche einen Chunk-unter-zu-bringen durchzuführen
        ChunkRIDs rids = push_rekursiv(moving_chunk, 0);

        chunks[move_idx].override_rids(rids);

        return &chunks[idx];
    }

    ChunkRIDs push_rekursiv(const Chunk& chunk, ushort number_of_try) {
        ChunkKey key = chunk.get_key();
        // key zu index auflösen
        size_t idx = hash(key);

        // merke hier nicht mehr den ersten index sondern "Anzahl an Versuchen" weiter, damit kein unendliches hin-und-her tauschen von 2 Elemetnen entsteht.
        size_t move_idx = idx;
        for (int i = 0; i <= number_of_try; ++i) {
            next_idx(number_of_try, move_idx, key);
        }
        // nutze loop um 'SEARCH_DEPTH' viele Kollisionen aufzulösen
        ushort count = 0;
        while (count < SEARCH_DEPTH) {
            if (chunks[idx].is_free()) {
                ChunkRIDs new_rids = chunks[idx].get_rids();
                if (chunk.get_voxels()) {
                    chunks[idx] << chunk.get_voxels();
                }
                chunks[idx] << key;
                chunks[idx].override_rids(chunk.get_rids());
                _occupancy += 1;
                return new_rids;
            }
            next_idx(count,idx,key);
        }
        // Wenn nach 'SEARCH_DEPTH' Kollisionen kein Platz gefunden wurde: Chunk austauschen und neu Versuchen (hierbei MÜSSEN RIDs getauscht werden!! )
        msg = "Nyoooo! Es hat schon WIEDER nicht geklappt OwO'";
        // Daten des zu überschreibenen Chunks kopieren
        Chunk moving_chunk{chunks[move_idx]};
        // Chunk, der keinen Platz gefunden hat, einsetzen
        chunks[move_idx] << chunk.get_voxels();
        chunks[move_idx] << key;
        chunks[move_idx].override_rids(chunk.get_rids());

        // Funktion wiederholen, jetzt mit herausgepflückten Chunk, zurück kommen die freien RIDs
        ChunkRIDs rids = push_rekursiv(moving_chunk, number_of_try);

        return rids;
    }
};

template <typename T>
class TaggedPointer {
    static_assert(8 != alignof(T), "type T für TaggedPointer ungeeignet.");
    uintptr_t ptr;
    enum FLAGS : uintptr_t {
        F1  = 0b001,
        F2  = 0b010,
        F3  = 0b100,
        ALL = 0b111
    };
    
public:
    TaggedPointer(T* obj_ptr = nullptr) : ptr(reinterpret_cast<uintptr_t>(obj_ptr)) {}

    bool get_f1() const {
        return ptr & F1;
    }
    void set_f1() {
        ptr |= F1;
    }
    void unset_f1() {
        ptr &= ~F1;
    }

    bool get_f2() const {
        return ptr & F2;
    }
    void set_f2() {
        ptr |= F2;
    }
    void unset_f2() {
        ptr &= ~F2;
    }
    

    T& operator*() {
        return *reinterpret_cast<T*>(ptr & ~ALL);
    }
    T* operator->() {
        return reinterpret_cast<T*>(ptr & ~ALL);
    }
    operator T *() {
        return reinterpret_cast<T*>(ptr & ~ALL);
    }

    const T& operator*() const {
        return *reinterpret_cast<const T*>(ptr & ~ALL);
    }
    const T* operator->() const {
        return reinterpret_cast<const T*>(ptr & ~ALL);
    }
    operator const T *() const {
        return reinterpret_cast<const T*>(ptr & ~ALL);
    }
};

	static constexpr ushort SEARCH_DEPTH = 13;
	static constexpr ushort TRYS = 20;
	size_t _occupancy = 0;
	
	std::array<Data, SIZE> data{};

	inline static size_t hash(const Key& key) {
		return Hash(key) % SIZE; // TODO: "Remainder by zero is undefined" vieleicht muss ich das verbessern, sollte aber eigentlich nie passieren
	}
	inline static void next_idx(ushort &count, size_t &idx, const Key &key) {
		count += 1;
		idx += count * 2 + key.x;
		idx -= SIZE * (SIZE <= idx);
	}

public:
	inline double get_occupancy_rate() {
		return static_cast<double>(_occupancy) / static_cast<double>(SIZE);
	}

	Data *get(const Key &key) {
		ushort count = 0;
		size_t idx = hash(key);
		while (count < SEARCH_DEPTH) {
			if (key == data[idx] && !data[idx].is_free()) {
				return &data[idx];
			}
			next_idx(count,idx,key);
		}
		return nullptr;
	}

	Data* remove(const Key &key) {
		Data *d = get(key);
		if (d) {
			_occupancy -= 1;
			d->free();
		}
		return d;
	}
	
	void push(Key &&key, typename Data::Content *content) {
		// TODO: dieser Teil erstmal ein workaround und zum sicher gehen:
		if (SIZE * 0.8 < _occupancy) {
			return; // mehr als 80% sollte nicht befüllt werden
		}
		// // überprüfen ob Chunk breits geladen ist
		// if (auto r = get(key)) {
		// 	return r; // TODO: sollte vor 'push' geprüft werden
		// }

		// austauschbare key Referenz erstellen
		const Key *_key = &key;
		Key moving_key{};
		// Die RIDs des neuen Platzes müssen mit dem des Verschobenen Tauschen
		Data* moving_data_ptr = nullptr;
		Data* original_data = nullptr;
		
		// nutze loop für 'TRYS' viele Versuche einen Chunk unter zu bringen
		ushort move_count = 0;
		while (move_count < TRYS) {
			// key zu index auflösen
			size_t idx = hash(*_key);
			// merke den ersten index, beim Scheitern wird diese Var. genutzt um das zuverschiebene Element zu bestimmen
			size_t move_idx = idx;
			// nutze loop um 'SEARCH_DEPTH' viele Kollisionen aufzulösen
			ushort count = 0;
			while (count < SEARCH_DEPTH) {
				if (data[idx].is_free()) {
					if (content) {
                        data[idx] << content;
					}
					data[idx] << key;
					_occupancy += 1;
					if (moving_data_ptr) {
						// Es wird ein Chunk verschoben, d.h. RIDs müssen übertragen werden
						original_data->override_rids(data[idx]);
						data[idx].override_rids(*moving_data_ptr);
					}
					return;
                }
                next_idx(count,idx,key);
            }
				
			// Wenn nach 'SEARCH_DEPTH' Kollisionen kein Platz gefunden wurde: Chunk austauschen und neu Versuchen

			// !!!!!!!!!!FEHLER!!!!!!!! Was ist mit den RIDs wenn getauscht wird?! RIDs müssen getauscht werden
			//TODO: chunk heraus nehmen. Beim ersten mal an der ersten stelle, danach an der 2. und immer so weiter bis 'TRYS' (...dann weiß ich nicht mehr weiter... sollt aber nie passieren *ahem* *hust* *hust*)
            // Daten des zu überschreibenen Chunks kopieren
			Data moving_data{data[move_idx]};
			moving_data_ptr = &moving_data;
			original_data = &data[move_idx];
			// Chunk, der keinen Platz gefunden hat, einsetzen (RIDs fehlen hier noch!)
			data[move_idx] << content;
            data[move_idx] << key;
			// Funktion wiederholen, jetzt mit herausgepflückten Chunk und neuem key
			moving_key << moving_data;
			_key = &moving_key;
			content = moving_data.data;

			// move_count hochzählen
			next_idx(move_count, move_idx, key);
		}
	}
};

template<typename Data, auto Func>
class WorkerThread {
    std::atomic_bool _terminate_thread = false;
    std::atomic_bool _is_pending = false;
    
	StrongThreadData<Data> inside;
	
	void load_and_remove_chunks();
	
public:
    std::thread thread;

    inline void start(WeakThreadData<Data> &outside) {
        if (thread.joinable()) {return;}
		run_again(outside);
        thread = std::thread([&]() {
            while (!_terminate_thread) {
                // Arbeiten
                Func(inside); // <- "heavy work"
				
                // gebe Daten wieder frei
                inside.dismiss(outside);

                // warte auf reaktivierung...
                do {
                    _is_pending = true;
                    while (_is_pending && !_terminate_thread) {
                        std::this_thread::sleep_for(std::chrono::milliseconds(WORKER_THREAD_ACTUALISATION_TIME_INTERVALL));
                    }
                    // ...und darauf, dass inside erreichbar ist
                } while (!inside && !_terminate_thread);
            }
            // thread terminiert
        });
    }

    inline void terminate_thread() {
        _terminate_thread = true;
        if (thread.joinable()) {
            thread.join();
        }
    }
    inline void run_again(WeakThreadData<Data> &outside) {
        // beschlagnahme daten
        inside.absorb(outside);
        // Vorab-Berechnung von Daten, die nur auf dem Main-Thread passieren können
        for (auto &cg : *inside) {
			// Node-Koordinaten berechnung
			Vector3 local_player_pos = LocalGameInstance::singleton->get_player()->get_position() - cg->body_data.node->get_position();
            // in chunks umrechnen
			local_player_pos /= (CHUNK_SIZE * VOXEL_SIZE);
            // 'floor' verwenden um negative koords richtig darzustellen (-0.5 -> -1) (0.5 -> 0)
			cg->set_local_player_koords(local_player_pos.floor());
        }
        // Thread aktive machen
        _is_pending = false;
    }
    inline bool is_ready_and_waiting() {
        return _is_pending;
	}
	
    ~WorkerThread() {terminate_thread();}
};

	return (c1.x == c2.x) && (c1.y == c2.y) && (c1.z == c2.z) && (c1.grid_id == c2.grid_id);
}

void VVG::Chunk::_create_face_xy(const ID block_id, const short x const short y, const const Vector2 &end, short z_offset, bool invert_normals) {
	
	_vertices.append(Vector3(begin.x, end.y  , z_offset) - OFFSET); // 0     0       1
	_vertices.append(Vector3(begin.x, begin.y, z_offset) - OFFSET);// 1
    _vertices.append(Vector3(end.x  , begin.y, z_offset) - OFFSET);// 2
    _vertices.append(Vector3(end.x  , end.y  , z_offset) - OFFSET);// 3     3       2

    const auto [ x, y ] = get_texture_coords(block_id);

	_uvs.append(Vector2(SCALE* x   , SCALE* y   ));
	_uvs.append(Vector2(SCALE*(x+1), SCALE* y   ));
    _uvs.append(Vector2(SCALE*(x+1), SCALE*(y+1)));
    _uvs.append(Vector2(SCALE* x   , SCALE*(y+1)));

	_add_polygons(invert_normals);
}
void VVG::Chunk::_create_face_yz(const ID block_id, const short x const short y, const const Vector2 &end, short x_offset, bool invert_normals) {
    _vertices.append(Vector3(x_offset, begin.x, end.y  ) - OFFSET);// 0     0       1
    _vertices.append(Vector3(x_offset, begin.x, begin.y) - OFFSET);// 1

}

bool Chunk::is_in_box(const Vector3i& begin,const Vector3i& end) {
	for (size_t i = 0; i < boxes.size(); ++i) {
		if (
            (boxes[i].begin.x <= begin.x) & (end.x <= boxes[i].end.x) &&
            (boxes[i].begin.y <= begin.y) & (end.y <= boxes[i].end.y) &&
            (boxes[i].begin.z <= begin.z) & (end.z <= boxes[i].end.z)
        ) {
			return true;
		}
	}
	return false;
}
